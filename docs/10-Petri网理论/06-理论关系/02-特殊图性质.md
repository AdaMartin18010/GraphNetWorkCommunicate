    """Petriç½‘å›¾ç»“æ„è¡¨ç¤º"""

    def __init__(self, places: Set[str], transitions: Set[str],
                 flow_relation: Set[tuple]):
        """

        Args:

            transitions: å˜è¿é›†åˆ

        self.adjacency = defaultdict(list)

        # æ„å»ºé‚»æ¥è¡¨
        for source, target in flow_relation:
            self.adjacency[source].append(target)

        """

            visited: å·²è®¿é—®é¡¶ç‚¹é›†åˆï¼ˆé€’å½’ä½¿ç”¨ï¼‰

            è®¿é—®åºåˆ—

            visited = set()

        visited.add(start)
        result = [start]

        for neighbor in self.adjacency[start]:
            if neighbor not in visited:
                result.extend(self.dfs(neighbor, visited))

        return result

    def dfs_iterative(self, start: str) -> List[str]:
        """
        è¿­ä»£å¼æ·±åº¦ä¼˜å…ˆæœç´¢

        Args:
            start: èµ·å§‹é¡¶ç‚¹

        Returns:
            è®¿é—®åºåˆ—

        stack = [start]
        result = []

            vertex = stack.pop()
            if vertex not in visited:
                visited.add(vertex)
                result.append(vertex)

                # å°†é‚»å±…å‹å…¥æ ˆï¼ˆé€†åºä»¥ä¿æŒé¡ºåºï¼‰
                for neighbor in reversed(self.adjacency[vertex]):

    def detect_cycles(self) -> List[List[str]]:

        æ£€æµ‹æ‰€æœ‰å¾ªç¯

        Returns:
            å¾ªç¯åˆ—è¡¨ï¼Œæ¯ä¸ªå¾ªç¯æ˜¯ä¸€ä¸ªé¡¶ç‚¹åºåˆ—

        visited = set()

            rec_stack.add(vertex)
            path.append(vertex)

                elif neighbor in rec_stack:

                    cycle_start = path.index(neighbor)
                    cycles.append(path[cycle_start:] + [neighbor])

            rec_stack.remove(vertex)
            path.pop()

        for vertex in self.vertices:
            if vertex not in visited:
                dfs_cycle(vertex, [])

        return cycles

        å¹¿åº¦ä¼˜å…ˆæœç´¢

        Args:
            start: èµ·å§‹é¡¶ç‚¹

        Returns:
            è®¿é—®åºåˆ—ï¼ˆæŒ‰BFSé¡ºåºï¼‰

                if neighbor not in visited:
                    visited.add(neighbor)

import heapq
from typing import Dict, Tuple, Optional

class WeightedPetriNetGraph:
    """å¸¦æƒé‡çš„Petriç½‘å›¾"""

    def __init__(self, places: Set[str], transitions: Set[str],
                 weighted_flow: Dict[Tuple[str, str], float]):
        """
        åˆå§‹åŒ–å¸¦æƒé‡çš„Petriç½‘å›¾

        Args:
            places: åº“æ‰€é›†åˆ
            transitions: å˜è¿é›†åˆ
            weighted_flow: å¸¦æƒé‡çš„æµå…³ç³»ï¼Œé”®ä¸º (source, target)ï¼Œå€¼ä¸ºæƒé‡
        """
        self.places = places
        self.transitions = transitions
        self.vertices = places | transitions
        self.adjacency = defaultdict(list)
        self.weights = {}

        for (source, target), weight in weighted_flow.items():
            self.adjacency[source].append(target)
            self.weights[(source, target)] = weight

    def dijkstra(self, start: str) -> Dict[str, Tuple[float, Optional[str]]]:
        """
        Dijkstraç®—æ³•è®¡ç®—å•æºæœ€çŸ­è·¯å¾„

        Args:
            start: èµ·å§‹é¡¶ç‚¹

        Returns:
            å­—å…¸ï¼Œé”®ä¸ºé¡¶ç‚¹ï¼Œå€¼ä¸º (æœ€çŸ­è·ç¦», å‰é©±é¡¶ç‚¹)
        """
        distances = {v: float('inf') for v in self.vertices}
        distances[start] = 0.0
        predecessors = {v: None for v in self.vertices}

        # ä¼˜å…ˆé˜Ÿåˆ—ï¼š(è·ç¦», é¡¶ç‚¹)
        pq = [(0.0, start)]
        visited = set()

        while pq:
            current_dist, current = heapq.heappop(pq)

            if current in visited:
                continue

            visited.add(current)

            for neighbor in self.adjacency[current]:
                edge_weight = self.weights.get((current, neighbor), 1.0)
                new_dist = current_dist + edge_weight

                if new_dist < distances[neighbor]:
                    distances[neighbor] = new_dist
                    predecessors[neighbor] = current
                    heapq.heappush(pq, (new_dist, neighbor))

        return {v: (distances[v], predecessors[v]) for v in self.vertices}

    def get_shortest_path(self, start: str, target: str) -> List[str]:
        """
        è·å–æœ€çŸ­è·¯å¾„

        Args:
            start: èµ·å§‹é¡¶ç‚¹
            target: ç›®æ ‡é¡¶ç‚¹

        Returns:
            æœ€çŸ­è·¯å¾„ï¼ˆé¡¶ç‚¹åºåˆ—ï¼‰
        """
        result = self.dijkstra(start)
        path = []
        current = target

        if result[target][0] == float('inf'):
            return []  # ä¸å¯è¾¾

        while current is not None:
            path.append(current)
            current = result[current][1]

        return list(reversed(path))

from typing import Dict, List

    """Petriç½‘å›¾ç»“æ„è¡¨ç¤ºï¼ˆç»­ï¼‰"""

    def floyd_warshall(self) -> Dict[Tuple[str, str], float]:
        """
        Floyd-Warshallç®—æ³•è®¡ç®—æ‰€æœ‰é¡¶ç‚¹å¯¹ä¹‹é—´çš„æœ€çŸ­è·¯å¾„

        # å¯¹è§’çº¿ä¸º0

            dist[i][i] = 0.0

        # åˆå§‹åŒ–ç›´æ¥è¿æ¥çš„è¾¹
        for source in self.vertices:
            for target in self.adjacency[source]:
                i = vertex_to_index[source]

        for k in range(n):
            for i in range(n):

                        dist[i][j] = dist[i][k] + dist[k][j]

        result = {}
        for i in range(n):
            for j in range(n):
                if dist[i][j] != float('inf'):
                    result[(vertices[i], vertices[j])] = dist[i][j]

        return result

class PetriNetGraph:
    """Petriç½‘å›¾ç»“æ„è¡¨ç¤ºï¼ˆç»­ï¼‰"""

    def tarjan_scc(self) -> List[List[str]]:
        """
        Tarjanç®—æ³•æ‰¾åˆ°æ‰€æœ‰å¼ºè¿é€šåˆ†é‡

        Returns:
            å¼ºè¿é€šåˆ†é‡åˆ—è¡¨ï¼Œæ¯ä¸ªåˆ†é‡æ˜¯ä¸€ä¸ªé¡¶ç‚¹åˆ—è¡¨
        """
        index = 0
        indices = {}
        lowlinks = {}
        stack = []
        on_stack = set()
        sccs = []

        def strongconnect(vertex: str):
            nonlocal index
            indices[vertex] = index
            lowlinks[vertex] = index
            index += 1
            stack.append(vertex)
            on_stack.add(vertex)

            for neighbor in self.adjacency[vertex]:
                if neighbor not in indices:
                    strongconnect(neighbor)
                    lowlinks[vertex] = min(lowlinks[vertex], lowlinks[neighbor])
                elif neighbor in on_stack:
                    lowlinks[vertex] = min(lowlinks[vertex], indices[neighbor])

            if lowlinks[vertex] == indices[vertex]:
                scc = []
                while True:
                    w = stack.pop()
                    on_stack.remove(w)
                    scc.append(w)
                    if w == vertex:
                        break
                sccs.append(scc)

        for vertex in self.vertices:
            if vertex not in indices:
                strongconnect(vertex)

        return sccs

class PetriNetGraph:
    """Petriç½‘å›¾ç»“æ„è¡¨ç¤ºï¼ˆç»­ï¼‰"""

    def get_reverse_graph(self) -> 'PetriNetGraph':
        """è·å–åå‘å›¾"""

            for target in self.adjacency[source]:
                reverse_flow.add((target, source))

        """
        Kosarajuç®—æ³•æ‰¾åˆ°æ‰€æœ‰å¼ºè¿é€šåˆ†é‡

                    dfs1(neighbor)

            if vertex not in visited:
                dfs1(vertex)

            scc.append(vertex)

                    dfs2(neighbor, scc)

        for vertex in reversed(finish_order):

                sccs.append(scc)
