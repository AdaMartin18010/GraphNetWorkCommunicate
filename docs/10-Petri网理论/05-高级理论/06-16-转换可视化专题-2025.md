class GraphvizVisualizer(Visualizer):
    """Graphvizå¯è§†åŒ–å™¨ï¼ˆå®šä¹‰2.1ï¼‰"""

    def visualize(self, model: Any) -> str:
        """

    def _visualize_fsm(self, fsm: Any, dot: graphviz.Digraph) -> str:
        """å¯è§†åŒ–FSMï¼ˆç®—æ³•2.1ï¼‰"""

        """å¯è§†åŒ–Petriç½‘"""

                    shape="circle", style="filled" if marking > 0 else "solid")

    def save(self, visualization: str, filename: str, format: str = "png"):
        """ä¿å­˜å¯è§†åŒ–"""

        dot.render(filename, format=format, cleanup=True)

class NetworkXVisualizer(Visualizer):
    """NetworkXå¯è§†åŒ–å™¨ï¼ˆå®šä¹‰3.1ï¼‰"""
