                        infected.add(i)
                        infected_times[i] = t

        """è®¡ç®—æ—¶åºå¯è¾¾æ€§"""

                        reachable[j] = t
                if j in reachable and reachable[j] <= t:
                    if i not in reachable or t < reachable[i]:
                        reachable[i] = t

        return target in reachable, reachable.get(target)
