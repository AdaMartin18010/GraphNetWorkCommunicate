            if (d - b) > self._threshold_persistence
        ]

        complexity = {
            'high_dimensional_features': len(high_dim_features),
            'persistent_features': len(persistent_features),
            'complexity_score': len(high_dim_features) + len(persistent_features)
        }

        return complexity
