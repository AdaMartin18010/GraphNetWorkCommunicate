### 步骤2: 批量移动文件

**使用Python脚本**:
```python
import os
import shutil
from pathlib import Path

# 定义文件映射
file_mapping = {
    # 项目推进报告
    "项目推进*.md": "archive/project-reports/completion-reports/2025-01/project-advancement/",
    "项目持续推进*.md": "archive/project-reports/completion-reports/2025-01/project-continuation/",
    "项目最终*.md": "archive/project-reports/completion-reports/2025-01/project-summary/",
    "项目完整*.md": "archive/project-reports/completion-reports/2025-01/project-summary/",
    "项目全面*.md": "archive/project-reports/completion-reports/2025-01/project-summary/",

    # 任务报告
    "P0任务*.md": "archive/project-reports/task-reports/2025-01/P0-tasks/",
    "P1任务*.md": "archive/project-reports/task-reports/2025-01/P1-tasks/",
    "P2任务*.md": "archive/project-reports/task-reports/2025-01/P2-tasks/",
    "P4任务*.md": "archive/project-reports/task-reports/2025-01/P4-tasks/",
    "P5任务*.md": "archive/project-reports/task-reports/2025-01/P5-tasks/",

    # 进展报告
    "*2025-12-05.md": "archive/project-reports/progress-reports/2025-12/",
    "任务推进*.md": "archive/project-reports/progress-reports/2025/",
    "阶段*.md": "archive/project-reports/progress-reports/2025/",
    "项目文件梳理*.md": "archive/project-reports/progress-reports/2025/",

    # 应用案例报告
    "应用案例*.md": "archive/project-reports/application-reports/2025-01/",
    "测试用例*.md": "archive/project-reports/application-reports/2025-01/",

    # 思维工具报告
    "思维工具*.md": "archive/project-reports/mind-tools-reports/2025-01/",
    "跨模块关联*.md": "archive/project-reports/mind-tools-reports/2025-01/",

    # 计划报告
    "*计划*.md": "archive/project-reports/planning-reports/2025/",
    "*报告*.md": "archive/project-reports/planning-reports/2025/",
}

# 执行归档
def archive_files():
    root = Path(".")
    for pattern, dest_dir in file_mapping.items():
        dest_path = Path(dest_dir)
        dest_path.mkdir(parents=True, exist_ok=True)

        # 匹配文件
        for file_path in root.glob(pattern):
            if file_path.is_file() and file_path.suffix == ".md":
                # 排除核心文档
                if file_path.name not in ["README.md", "CONTRIBUTING.md", "PROJECT_COMPLETION.md"]:
                    shutil.move(str(file_path), str(dest_path / file_path.name))
                    print(f"Moved: {file_path.name} -> {dest_dir}")

if __name__ == "__main__":
    archive_files()
```
